Install ax5

ax5 시작하기

ax5라이브러리 파일을 html head안에 insert 합니다.

<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8"/>
        <title>ax5</title>
    </head>
    <body>

    </body>
</html>

ax5라이브러리 파일을 html head안에 insert 합니다.

<head>
    <meta charset="UTF-8"/>
    <title>ax5</title>
    <u><script type="text/javascript" src="pub/ax5.min.js"></script></u>
</head>

이렇게 준비된 html파일에 script 태그를 페이지 최 하단에 삽입하고 원하는 코드를 타이핑 합니다. 

<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8"/>
        <title>ax5</title>
    </head>
    <body>

    <script>
    var util = ax5.util, dom = ax5.dom;
    // 개발에 편의상 ax5.util을 util로 바꾸어 코딩하는 것을 권장합니다. util이라는 변수를 기존에 사용하지 않는 경우입니다.
    console.log( ax5.info.url_util() );
    
    </script>
    </body>
</html>

ax5.util.alert

JSON alert

<button data-button="ax5-util-alert-0">{a:1, b:2}</button>
<script>
    ax5.dom("[data-button='ax5-util-alert-0']").on("click", function(){
        ax5.util.alert({a:1, b:2});
    });
</script>

아래의 버튼을 클릭하시면 alert이 출력됩니다.

ax5.xhr

일반적인 방법

ax5.xhr{
    method: "POST",
    url   : "data.php",
    param : "str=1234",
    res   : function (response, status) {
        console.log("success");
        console.log(this);
    },
    error : function () {
        console.log("error");
        console.log(this);
    }
});

error 를 사용하지 않는 방법

ax5.xhr({ // req 또는 request 둘다 사용 가능합니다.
    method: "POST",
    url   : "data.php",
    param : "str=1234",
    res   : function (response, status) {
        // status 값이 200 인지 판단 가능
        console.log(this);
    }
});

timeout 값 설정

ax5.xhr({
    method: "POST",
    url   : "data.php",
    param : "str=1234",
    res   : function (response, status) {
        // status 값이 200 인지 판단 가능
        console.log(this);
    },
    timeout: 1
});
