ax5라이브러리 파일을 html head안에 insert 합니다.

<head>
    <meta charset="UTF-8"/>
    <title>ax5</title>
    <u><script type="text/javascript" src="pub/ax5.min.js"></script></u>

ax5.util.clone

<div id="ax5-util-clone" class="script-result"></div>
<script>
    (function(){
        var a = {a:1};
        var b = a;
        var el = ax5.dom("#ax5-util-clone");

        el.append("(a == b) => " + (a == b) + "<br/>");
        b = ax5.util.clone(a);
        el.append("(a == b) => " + (a == b));
    })();
</script>

Result

변수 a를 b에 대입하면 변수 b는 a의 참조가 됩니다. 참조가 된 변수 b의 속성 b.a의 값을 바꾸었을 때 a.a의 값도 함께 바뀌는 상황을 초래합니다. 그런 이유로 때로 원치 않는 결과를 얻을 수 있습니다. 이럴 때엔 ax5.util.clone함수를 이용하여 변수 a의 복제된 오브젝트를 만들어 사용할 수 있습니다.

ax5.util.each

<div id="ax5-util-each" class="script-result"></div>
<script>
    (function(){
        var util = ax5.util;
        var a = {a:"axisj", b:"ax5"};
        var b = [9,8,5,4,3,1];
        var el = ax5.dom("#ax5-util-each");

        var po = [];
        po.push("* a");
        util.each(a, function(k, v){
            po.push(k + ":" + v + " | " + this);
        });
        po.push("* b");
        util.each(b, function(index, item){
            po.push(index + ":" + item + " | " + this);
        });
        el.html(po.join('<br/>'));
    })();
</script>

Result

each를 이용하여 Array또는 Object의 아이템을 순환처리 할수 있습니다.

ax5.util.search

<div id="ax5-util-search" class="script-result"></div>
<script>
    (function(){
        var util = ax5.util, el = ax5.dom("#ax5-util-search");
        var a = ["A","X","5"];
        var idx = util.search(a, function(){
            return this == "X";
        });
        el.append(a[idx]);
        el.append(" / ");
        el.append(a[
                util.search(a, function(idx){
                    return idx == 2;
                })
            ]);

        var b = {a:"AX5-0", x:"AX5-1", 5:"AX5-2"};
        el.append(" / ");
        el.append(b[
                util.search(b, function(k, v){
                    return k == "x";
                })
            ]);
    })();
</script>

Result

search는 배열을 대상으로 할 때와 오브젝트를 대상으로 할 때 결과가 다르다. Array의 경우에는 Array의 인덱스가 리턴되고 Object의 경우에는 Object의 키가 리턴된다.

ax5.util.reduce

<div id="ax5-util-reduce" class="script-result"></div>
<script>
    (function(){
        var util = ax5.util, el = ax5.dom("#ax5-util-reduce");
        var aarray = [5,4,3,2,1], result;
        result = util.reduce( aarray, function(p, n){
            return p * n;
        });
        el.append('result = ' + result);
        el.append('/' + util.to_json(aarray) );

        result = util.reduce({a:1, b:2}, function(p, n){
            // Object의 경우 첫번째 p 값이 undefined 입니다.
            return parseInt(p|0) + parseInt(n);
        });
        el.append('<br/>result = ' + result);
    })();
</script>

Result

배열의 왼쪽에서 오른쪽으로 연산을 진행하는데 수행한 결과가 왼쪽 값으로 반영되어 최종 왼쪽 값을 반환합니다.

ax5.xhr

일반적인 방법

    ax5.xhr{
        method: "POST",
        url   : "data.php",
        param : "str=1234",
        res   : function (response, status) {
            console.log("success");
            console.log(this);
        },
        error : function () {
            console.log("error");
            console.log(this);
        }
    });

error 를 사용하지 않는 방법

    ax5.xhr({ // req 또는 request 둘다 사용 가능합니다.
        method: "POST",
        url   : "data.php",
        param : "str=1234",
        res   : function (response, status) {
            // status 값이 200 인지 판단 가능
            console.log(this);
        }
    });

timeout 값 설정

    ax5.xhr({
        method: "POST",
        url   : "data.php",
        param : "str=1234",
        res   : function (response, status) {
            // status 값이 200 인지 판단 가능
            console.log(this);
        },
        timeout: 1
    });

queue 처리

     var xhr_queue = [];
     xhr_queue.push({
        method: "POST",
        url   : "../samples/ax5/xhr/data.php",
        param : "str=1234",
        res   : function (response, status) {
            console.log(response);
        }
     });
     xhr_queue.push({
        method: "POST",
        url   : "../samples/ax5/xhr/data.php",
        param : "str=1234",
        res   : function (response, status) {
            console.log(response);
        }
     });
