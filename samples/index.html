    ax5.util.alert

    JSON alert

    <button data-button="ax5-util-alert-0">{a:1, b:2}</button>
<script>
    ax5.dom("[data-button='ax5-util-alert-0']").on("click", function(){
        ax5.util.alert({a:1, b:2});
    });
</script>

    ax5.xhr

    일반적인 방법

    ax5.xhr{
    method: "POST",
    url   : "data.php",
    param : "str=1234",
    res   : function (response, status) {
        console.log("success");
        console.log(this);
    },
    error : function () {
        console.log("error");
        console.log(this);
    }
});

    error 를 사용하지 않는 방법

    ax5.xhr({ // req 또는 request 둘다 사용 가능합니다.
    method: "POST",
    url   : "data.php",
    param : "str=1234",
    res   : function (response, status) {
        // status 값이 200 인지 판단 가능
        console.log(this);
    }
});

    timeout 값 설정

    ax5.xhr({
    method: "POST",
    url   : "data.php",
    param : "str=1234",
    res   : function (response, status) {
        // status 값이 200 인지 판단 가능
        console.log(this);
    },
    timeout: 1
});
