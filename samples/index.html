<!DOCTYPE html>
<html lang="en">
    <head>

        <title>ax5</title>
    </head>

ax5라이브러리 파일을 html head안에 insert 합니다.

<head>
    <meta charset="UTF-8"/>
    <title>ax5</title>
    <u><script type="text/javascript" src="pub/ax5.min.js"></script></u>
</head>

이렇게 준비된 html파일에 script 태그를 페이지 최 하단에 삽입하고 원하는 코드를 타이핑 합니다.

ax5.util

ax5.util.get_type

get_type이 지원하는 자료형

number, string, array, object, element, nodelist, function, undefined, fragment를 지원합니다.
ax5.dom.get("#ax5-util-get-type")를 했을 때. nodelist가 아닌 array가 반환되는데. 이는 ax5내부에서 nodelist를 array로 변환하여 관리하고 있기 때문이다. 

<div id="ax5-util-get-type" class="script-result"></div>
<script>
    (function(){
        var util = ax5.util, result, el = ax5.dom("#ax5-util-get-type");

        el.append( "1 = " + util.get_type(1) + "<br/>" );
        el.append( "'1' = " + util.get_type("1") + "<br/>" );
        el.append( "[0,1,2] = " + util.get_type([0,1,2]) + "<br/>" );
        el.append( "{a:1} = " + util.get_type({a:1}) + "<br/>" );
        el.append( 'ax5.dom.get("#ax5-util-get-type")[0] = '
        + util.get_type( ax5.dom.get("#ax5-util-get-type")[0] ) + "<br/>" );
        el.append( 'document.querySelectorAll(".content") = '
        + util.get_type( document.querySelectorAll(".content") ) + "<br/>" );
        el.append( 'function(){} = ' + util.get_type( function(){} ) + "<br/>" );

        var ___a;
        el.append( 'var ___a; ___a = ' + util.get_type( ___a ) + "<br/>" );
        el.append( 'document.createDocumentFragment() = '
        + util.get_type( document.createDocumentFragment() ) + "<br/>" );
    })();
</script>

ax5.util.is_[type]

argument타입에 따라 다양한 기능을 구현하거나 또는 보다 유연한 타입의 함수 개발을 위해서는 변수 자료형에 대한 타입체크가 필수 이다. ax5에서는 다음의 타입에 대해 is_메소드를 지원한다.

    // 오브젝트가 window인지 체크 합니다.
    ax5.util.is_window( window );

    // 오브젝트가 element인지 체크 합니다.
    ax5.util.is_element( ax5.dom.get("#ax5-util-is-type")[0] );

    // 오브젝트가 Object인지 체크 합니다.
    ax5.util.is_object();

    // 오브젝트가 Array인지 체크 합니다.
    ax5.util.is_array();

    // 오브젝트가 Functon인지 체크 합니다.
    ax5.util.is_function();

    // 오브젝트가 String인지 체크 합니다.
    ax5.util.is_string();

    // 오브젝트가 Number인지 체크 합니다.
    ax5.util.is_number();

    // 오브젝트가 nodeList인지 체크 합니다.
    ax5.util.is_nodelist( document.querySelectorAll(".content") );

    // 오브젝트가 undefined인지 체크 합니다.
    ax5.util.is_undefined();

    // 오브젝트가 undefined|''|null 인지 체크 합니다.
    ax5.util.is_nothing();

ax5.util.clone

<div id="ax5-util-clone" class="script-result"></div>
<script>
    (function(){
        var a = {a:1};
        var b = a;
        var el = ax5.dom("#ax5-util-clone");

        el.append("(a == b) => " + (a == b) + "<br/>");
        b = ax5.util.clone(a);
        el.append("(a == b) => " + (a == b));
    })();
</script>

Result

변수 a를 b에 대입하면 변수 b는 a의 참조가 됩니다. 참조가 된 변수 b의 속성 b.a의 값을 바꾸었을 때 a.a의 값도 함께 바뀌는 상황을 초래합니다. 그런 이유로 때로 원치 않는 결과를 얻을 수 있습니다. 이럴 때엔 ax5.util.clone함수를 이용하여 변수 a의 복제된 오브젝트를 만들어 사용할 수 있습니다.

ax5.util.reduce

<div id="ax5-util-reduce" class="script-result"></div>
<script>
    (function(){

        var aarray = [5,4,3,2,1], result;

ax5.util.extend

타겟 오브젝트에 원하는 오브젝트를 덮어쓰기 한다. 덮어쓰기 옵션으로 마지막 인자로 true를 부여 하면 덮어쓰기 처리된다. 예제에서와 같이 obj1.b의 경우 값이 치환되는 현상이 있는데. 하위 키까지 오브젝트를 덮어쓰기 하려면 extend_all을 사용한다. 

<div id="ax5-util-extend" class="script-result"></div>
<script>
    (function(){

        var po = [];

        var obj2 = {a:2, b:3, c:4};

        util.extend(obj1, obj2);
        po.push( util.to_json(obj1) );
        util.extend(obj1, obj2, true);

    })();
</script>

ax5.util.parse_json

json은 이제 xml을 밀어내고 범용 데이터 포멧의 표준이 되었다. 그런데 json데이터의 형식은 몇가지 규칙이 있는데. "key":"value"의 형식을 유지해야 한다는 점이다. key값을 따옴표로 감싸지 않으면 대게의 json parser는 에러는 뱉어내게 된다. 물론 대부분의 상황에서 규칙을 지켜야 겠지만. 귀찬을 때가 있다. 또는 function을 부여 한다던지. 

<div id="ax5-util-parse_json" class="script-result"></div>
<script>
(function(){
	var util = ax5.util, el = ax5.dom("#ax5-util-parse_json");
	var po = [];

	po.push( util.to_json( util.parse_json("[{'a':'99'},'2','3']")[0] ) );
	// ax5 parse_json은 json 구문에 대해 관대 합니다.
	po.push( util.parse_json("{a:1}").a );
	// 옵션 : 강제 적용 여부 (json 문자열 검사를 무시하고 오브젝트 변환을 시도합니다.)
	po.push( util.to_json( util.parse_json("{'a':1, 'b':function(){return 1;}}", false) ) );
	po.push( util.to_json( util.parse_json("{'a':1, 'b':function(){return 1;}}", true) ) );

	el.append( po.join('<br/>') );
})();
</script>

Result

ax5.dom

ax5.dom은 jQuery방식의 dom select와 네이티브 방식의 select를 혼용하여 사용 할 수 있게 설계되었다. 예를 들어 ax5.dom("#id")로 엘리먼트 select를 해둔 상태에서 ax5.dom에 준비된 다양한 method들을 사용하다가 for구문 안에서 element를 다룰 때에는 ax5.dom.css(element, {width:100});처럼 select를 다시 하시 않고 좀더 가볍고 직접적인 방법으로 원하는 기능을 사용하는 것이다.

ax5.dom() vs ax5.dom.get()

  • ax5.dom("#id").elements[0] === ax5.dom.get("#id")[0]
  • ax5.dom("#ax5-dom-get").width( ); === ax5.dom.width( ax5.dom.get("#ax5-dom-get") );
  • ax5.dom("#ax5-dom-get").attr("id"); === ax5.dom.attr( ax5.dom.get("#ax5-dom-get"), "id" );

얼핏 보면 ax5.dom.을 사용하는 것은 대단히 불편해 보인다. 하지만 프로그램의 성능을 위해서는 element select횟수를 최대한 줄이고 캐시된 element를 이용하여 코딩하려는 노력이 필요하다. 그리고 코드의 길이가 문제라면면각자의 개발환경에 맞도록 var util = ax5.util과 같은 방식으로 개인의 기호에 맞게 바꾸어 사용하면 크게 문제될 일도 아니다.

ax5.dom.get

CSS셀렉터가 복잡하고 다양한 버전의 Selecting Elements를 지원하는 것을 중요하다. 하지만 그 보다 중요한것은 속도이다. 복잡한 형태의 CSS Query문을 처리 하기위해 많은 리소스를 사용하는 것은 바람직하지 않다. ax5의 dom Selecting은 프로그래머가 보다 빠르게 Selecting Elements를 할 수 있는 제한적인 형태의 CSS Query 사용을 지향한다.

가급적 하나의 조건으로 Query하여 프로그램 하도록 유도하고 불가피하게 하나의 조건으로 쿼리 하지 못한다면 sub_query하여 Selecting 할 수 있도록 한다. 

<div id="ax5-dom-get" class="script-result">
	<form>
		<input type="text" class="ax-inp" />
		<button class="ax-btn">OK</button>
	</form>
</div>
<script>
(function(){
	var util = ax5.util, el = ax5.dom("#ax5-dom-get");
	var po = [];
	el.append( '<br/>' );
	ax5.dom.get("a").forEach(function(item){
		po.push( "" + item.tagName + " href=" + ax5.dom.attr(item, "href") );
	});
	el.append( po.join(',') );
	el.append( '<br/>' );

	po = [];
	ax5.dom.get(".visual").forEach(function(item){
		po.push( "" + item.tagName  + " class=" + ax5.dom.attr(item, "class") );
	});
	el.append( po.join(',') );
	el.append( '<br/>' );

	po = [];
	el.find("[type=text]").elements.forEach(function(item){
		po.push( "" + item.tagName  + " class=" + ax5.dom.attr(item, "class") );
	});
	el.append( po.join(',') );
	el.append( '<br/>' );

	po = [];
	ax5.dom.get("#ax5-dom-get", ".ax-btn").forEach(function(item){
		po.push( "" + item.tagName  + " class=" + ax5.dom.attr(item, "class") );
	});
	el.append( po.join(',') );
	el.append( '<br/>' );
})();
</script>

ax5.dom.class_name

jQuery방식으로 개발한다면 addClass, removeClass, toggleClass가 익숙하다. 필자도 옥돔의 방식으로 개발하면 가끔 헤깔린다. 그렇지만 api도 원칙에 입각해서 만들어 져야 한다. 옥돔의 원칙은 하나의 메소드로 여러가지 일을 하는 것이다.
다른 메소드들과 다르게 첫번째 인자로 add,remove,toggle,has명령어를 넣어야 한다. 명령어를 생략하면 has로 처리한다. class_name이 다른 다른 메소들과 다른 이유는 key, value 형식의 조건이 아니기 때문이다. (사실 만들면서 짜증이 났던 부분이다.) 

<div id="ax5-dom-class_name" class="script-result">
	<div class="target-div"></div>
</div>
<script>
(function(){
	var util = ax5.util, el = ax5.dom("#ax5-dom-class_name");
	var po = [];
	var target_div = el.find(".target-div");

	// 한개의 class 삽입
	target_div.class_name("add", "c");
	po.push( el.html().replace(/</g,'<') );

	// 여러개의 class 삽입
	ax5.dom.class_name(target_div, "add", ["a","b"]);
	// 이런식으로도 할 수 있다는건 이제 모두 잘 아는 사실이다. 모르겟다면 위로 스크롤 해보자
	po.push( el.html().replace(/</g,'<') );

	// toggle
	target_div.class_name("toggle", "a");
	po.push( el.html().replace(/</g,'<') );

	// 제거
	target_div.class_name("remove", "c");
	po.push( el.html().replace(/</g,'<') );

	el.append( po.join('<br/>') );
})();
</script>
