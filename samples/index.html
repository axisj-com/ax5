ax5.util

ax5.util.alert

<button data-button="ax5-util-alert-0">{a:1, b:2}</button>
<script>
    ax5.dom("[data-button='ax5-util-alert-0']").on("click", function(){
        ax5.util.alert({a:1, b:2});
    });
</script>

아래의 버튼을 클릭하시면 alert가 출력됩니다.

ax5.util.clone

<div id="ax5-util-clone" style="padding:10px;border: 1px solid;"></div>
<script>
    (function(){
        var a = {a:1};
        var b = a;
        var el = ax5.dom("#ax5-util-clone");

        el.append("(a == b) => " + (a == b) + "<br/>");
        b = ax5.util.clone(a);
        el.append("(a == b) => " + (a == b));
    })();
</script>

Result

변수 a를 b에 대입하면 변수 b는 a의 참조가 됩니다. 참조가 된 변수 b의 속성 b.a의 값을 바꾸었을 때 a.a의 값도 함께 바뀌는 상황을 초래합니다. 그런 이유로 때로 원치 않는 결과를 얻을 수 있습니다. 이럴 때엔 ax5.util.clone함수를 이용하여 변수 a의 복제된 오브젝트를 만들어 사용할 수 있습니다.

ax5.util.each

<div id="ax5-util-each" style="padding:10px;border: 1px solid;"></div>
<script>
    (function(){
        var util = ax5.util;
        var a = {a:"axisj", b:"ax5"};
        var b = [9,8,5,4,3,1];
        var el = ax5.dom("#ax5-util-each");

        var po = [];
        po.push("* a");
        util.each(a, function(k, v){
            po.push(k + ":" + v + " | " + this);
        });
        po.push("* b");
        util.each(b, function(index, item){
            po.push(index + ":" + item + " | " + this);
        });
        el.html(po.join('<br/>'));
    })();
</script>

Result

each를 이용하여 Array또는 Object의 아이템을 순환처리 할수 있습니다.

ax5.xhr

일반적인 방법

    ax5.xhr{
        method: "POST",
        url   : "data.php",
        param : "str=1234",
        res   : function (response, status) {
            console.log("success");
            console.log(this);
        },
        error : function () {
            console.log("error");
            console.log(this);
        }
    });

error 를 사용하지 않는 방법

    ax5.xhr({ // req 또는 request 둘다 사용 가능합니다.
        method: "POST",
        url   : "data.php",
        param : "str=1234",
        res   : function (response, status) {
            // status 값이 200 인지 판단 가능
            console.log(this);
        }
    });

timeout 값 설정

    ax5.xhr({
        method: "POST",
        url   : "data.php",
        param : "str=1234",
        res   : function (response, status) {
            // status 값이 200 인지 판단 가능
            console.log(this);
        },
        timeout: 1
    });

queue 처리

     var xhr_queue = [];
     xhr_queue.push({
        method: "POST",
        url   : "../samples/ax5/xhr/data.php",
        param : "str=1234",
        res   : function (response, status) {
            console.log(response);
        }
     });
     xhr_queue.push({
        method: "POST",
        url   : "../samples/ax5/xhr/data.php",
        param : "str=1234",
        res   : function (response, status) {
            console.log(response);
        }
     });

     ax5.xhr(xhr_queue, function() {
        console.log("큐 완료")
     });
