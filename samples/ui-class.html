UI Class

AX5 UI 클래스의 사용법과 상세 예제를 만나실 수 있습니다.

Install

ax5 ui 는 require 이용하여 개별 UI를 선택적으로 사용하는 방법과 ui/all.js를 이용하여 바로 사용하는 방법이 있다. 본 문서에서는 편의상 ui/all.js를 html head에 추가하여 소개하고 있다. 

<script type="text/javascript" src="../pub/ax5.js"></script>
<script type="text/javascript" src="../pub/ui/all.js"></script>

<script type="text/javascript">
ax5.info.base_url = "../pub/";
ax5.util.require(["ui/menu.js"], function() {

});
</script>

ax5.ui.**

ax5.ui는 core 클래스인 ax5.ui.root만 ax5.js파일에 포함되어 있고 나머지 각각의 UI 클래스들은 별도의 파일에 만들어져 있다. UI클래스들은 해당 UI를 사용하고자 할 때 ax5.util.require를 이용하여 스크립트 로딩하거나 HTML head에 삽입한 후 사용해야 한다.

본 페이지 예제에서는 모두 스크립트 로딩방식으로 설명하고 있지만 실제 업무에 적용할 때에는 각자 환경에 맞는 방법으로 적용할 수 있다.

Mask

웹 애플리케이션 전체또는 특정 영역을 불투명한 레이어로 덮어서 웹애플리케이션의 UI를 사용 불능 상태로 만들거나 비동기 데이터 호출중에 사용자에게 답답함을 줄여주는 역할을 합니다.

Element Mask

<div class="tag-result">
	<div id="mask-target" style="border:1px solid #ccc;width:500px;height:500px;text-align: center;">
		<h1>Mask Target</h1>
		<p>마스크 될 엘리먼트 타겟</p>
	</div>
	<div class="DH10"></div>
	<button class="ax-btn" onclick="app_mask_open();">마스크열기</button>

</div>
<script type="text/javascript">
(function(){
	window.app_mask = new ax5.ui.mask();
	app_mask.set_config({
		target: document.body, // 미리 선언했지만
		content: "<h1>ready</h1>",
		onchange: function(){

		}
	});

	window.app_mask_open = function() {
		app_mask.open({
			// 변경가능하다.
			target: ax5.dom.get("#mask-target"),
			content: "<h1>Loading..</h1>",
			onchange: function () {

			}
		});

		// 다음의 구문을 실행하면 mask_target은 set_config에서 설정한 target으로 처리 된다.
		// app_mask.open();
	};
	window.app_mask_close = function(){
		app_mask.close();
	};
})();
</script>

마스크 컨셉

마스크는 두가지 방법으로 사용할 수 있게 제작하였다. 첫째는 특정 엘리먼트에 마스크가 입혀지는 경우. 둘째는 웹브라우저내 문서 전체에 마스크가 입혀지는 것이다.
mask_target값으로 방식을 달리 할수 있다. document.body이나 ax5.dom.get("#mask-target")을 mask_target에 할당 할 수 있고. document.getElementById를 이용하여도 무관하다. 엘리먼트에 마스크가 입혀지는 경우 대상 엘리먼트에 .ax-masking클래스가 추가되면서 blur효과가 추가된다. 바디에 적용하는 경우에도 onchange 함수에서 this.type으로 예외처리 하여 원하는 대상에 .ax-masking 클래스를 추가/해제할 수 있다.

<script type="text/javascript">
(function(){
	var my_disp = ax5.dom.get_one("#ui-keypad-example-disp");
	var my_keypad = new ax5.ui.keypad();
	my_keypad.set_config({
		target: ax5.dom.get("#ui-keypad-example-target"),
		theme: "default",
		board: {
			keys: [
				{label:"7", value:7},
				{label:"8", value:8},
				{label:"9", value:9},
				{label:"0", value:0},
				{newline:true},
				{label:"4", value:4},
				{label:"5", value:5},
				{label:"6", value:6},
				{label:'<i class="axi axi-delete"></i>', value:"del", klass:"warning"},
				{newline:true},
				{label:"1", value:1},
				{label:"2", value:2},
				{label:"3", value:3},
				{label:"ENTER", value:"return", klass:"basic"}
			],
			col_width: "25%",
			col_height: "60px"
		},
		onclick: function(){
			if(this.item.value == null) return;
			if(this.item.value == "del"){
				if(my_disp.value == "") return;
				my_disp.value = ax5.util.number(ax5.util.number(my_disp.value.substr(0, my_disp.value.length-1)), {money:true});
			}
			else
			if(this.item.value == "return"){
				if(my_disp.value == "") return;
				alert(ax5.util.number(my_disp.value));
			}
			else{
				if(my_disp.value == "" && ax5.util.number(this.item.value) == 0) return;
				my_disp.value = ax5.util.number('' + ax5.util.number(my_disp.value) + this.item.value, {money:true});
			}
		}
	});
})();
</script>

Dialog

dialog는 window.alert나 window.confirm의 대체 수단으로 사용이 가능한 UI 이다. 어떻게 하면 잘 만들까? 생각을 하다보니 prompt도 구현해야 겠다.
dialog는 ax5.ui.mask합성하므로 해당 스크립트가 문서에 포함되어야 한다.

alert

alert는 단순 알림외에도 웹 브라우저나 웹 애플리케이션의 프로세스를 일시 중지 시키는 기능도 가지고 있다. 하지만 dialog에서는 개발자가 다소 어려운 방법으로 구현을 해야 가능하다. (아쉽다)

<div class="tag-result">
	<button class="ax-btn" id="ax-ui-dialog-alert-btn">Alert</button>
</div>
<script type="text/javascript">
(function(){
	// ui dependency : mask
	var my_dialog = new ax5.ui.dialog();
	my_dialog.set_config({
		title: '<i class="axi axi-ion-alert"></i> Default alert',
		mask: {} // default target : document.body
	});

	// default
	ax5.dom("#ax-ui-dialog-alert-btn").on("click", function(){
		my_dialog.alert(
		'<span class="ax">알러트 메세지 기본 타입의 Alert입니다.</span>\n<span class="ax basic">테마를 정하지 않았습니다.</span>',
		function(){
			console.log(this);
		});
	});
})();
</script>

Result

callback

전통적인 방식의 개발에서는 window.alert를 호출하면 웹 애플리케이션이 hold된다. 하지만 직접구현한 UI 컴포넌트에서 그런 기능을 기대 하는 것은 어리석다. 한번 더 생각해보면 자바스크립트로 그런 작업을 할 수 있다면 보안에 문제가 많기 때문이다. 스크립트를 삽입해서 웹 애플리케이션을 먹통으로 만드는 일이 너무도 빈번 하지 않겠는가. 그런 이유로 직접 구현한 UI 컴포넌트에서 dialog를 호출하고 사용자 입력 후 다음 함수를 실행 하게 하려면 callback함수를 이용하여야 한다.

my_dialog.alert({}, function(){});에서 두번째 인자가 callback함수가 되는데 사용자 입력을 받으면 호출되게 된다.

Toast

toast는 웹 애플리케이션의 상태를 사용자에게 세련되게 알려주는 UI이다. 스크린의 우측이나 우 하단 영역에서 박스가 노출되었다가 일정 시간이 지나면 자연스럽게 사라진다. 웹 애플리케이션에서는 웹 브라우저의 우측이나 우 하단 영역에서 박스가 노출 되었다가 사라지게 된다.

default
