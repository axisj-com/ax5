<div class="tag-result" style="height: 250px;">
	<div style="padding:3px;">
		<input type="text" class="ax-inp lg" id="ui-keypad-example-disp" value="" style="width:234px;text-align: right;" />
	</div>
	<div id="ui-keypad-example-target" style="width:240px;"></div>

<script type="text/javascript">
(function(){
	var my_disp = ax5.dom.get_one("#ui-keypad-example-disp");
	var my_keypad = new ax5.ui.keypad();
	my_keypad.set_config({
		target: ax5.dom.get("#ui-keypad-example-target"),
		theme: "default",
		board: {
			keys: [
				{label:"7", value:7},
				{label:"8", value:8},
				{label:"9", value:9},
				{label:"0", value:0},
				{newline:true},
				{label:"4", value:4},
				{label:"5", value:5},
				{label:"6", value:6},
				{label:'<i class="axi axi-delete"></i>', value:"del", klass:"warning"},
				{newline:true},
				{label:"1", value:1},
				{label:"2", value:2},
				{label:"3", value:3},
				{label:"ENTER", value:"return", klass:"basic"}
			],
			col_width: "25%",
			col_height: "60px"
		},
		onclick: function(){
			if(this.item.value == null) return;
			if(this.item.value == "del"){
				if(my_disp.value == "") return;
				my_disp.value = ax5.util.number(ax5.util.number(my_disp.value.substr(0, my_disp.value.length-1)), {money:true});
			}
			else
			if(this.item.value == "return"){
				if(my_disp.value == "") return;
				alert(ax5.util.number(my_disp.value));
			}
			else{
				if(my_disp.value == "" && ax5.util.number(this.item.value) == 0) return;
				my_disp.value = ax5.util.number('' + ax5.util.number(my_disp.value) + this.item.value, {money:true});
			}
		}
	});
})();
</script>

bigsize

아이템의 갯수가 많아지면 문서에 html노드를 너무 많이 만들어 내게 된다. 당장 사용하지도 않을 것인데. 그로 인데 웹 애플리케이션의 속도가 저하되는 문제가 발생된다. 이를 해결 하기 위해 현재 스크롤에 보이는 아이템만 출력시키는 꼼수가 필요하다.

아래 예제에서보면 list에 10,000개의 아이템이 담겨 있다. 하지만 div안에 담겨있는 html노드 수는 6개를 유지하고 있다. (헤드에 1개 바디에 5개) 

<div class="tag-result">
	<div id="ax-ui-touch-grid-bigsize-target" style="height:220px;"></div>
	<div style="padding:5px;">
		TR 의 갯수는 <span id="ax-ui-touch-grid-bigsize-tr-count"></span>
	</div>
</div>
<script type="text/javascript">
(function(){
	var my_touch_grid = new ax5.ui.touch_grid();
	my_touch_grid.set_config({
		target: ax5.dom.get("#ax-ui-touch-grid-bigsize-target"),
		head_height: 30,
		head_align: "center",
		item_height: 40,
		col_group: [
			{key: 'no', label: '번호', width: 120, align:"center", formatter: function(){
				return ax5.util.number(this.value, {money: true});
			}},
			{key: 'title', label: '제목', width: '*'},
			{key: 'price', label: 'Price', width: 120, formatter:"money", align:"right"}
		],
		body: {
			onclick: function(){
				console.log(this);
			}
		},
		control:{
			width: 50,
			klass: '',
			//first: '<i class="axi axi-angle-double-up"></i>',
			prev: '<i class="axi axi-angle-up"></i>',
			next: '<i class="axi axi-angle-down"></i>',
			//last: '<i class="axi axi-angle-double-down"></i>',
			onclick: function(){
				console.log(this);
				print_html_node();
			}
		}
	});

	var list = [];
	for(var i=1;i<10001;i++){
		list.push({no:i, title:'제목 '+ i,  price: i*1000});
	}
	my_touch_grid.set_list(list);

	function print_html_node(){
		ax5.dom('#ax-ui-touch-grid-bigsize-tr-count').html( '<b>' + ax5.dom("#ax-ui-touch-grid-bigsize-target").find("tr").elements.length + '개</b>' );
	}
})();
</script>
